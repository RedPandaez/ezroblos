	local function UpdateTools()
		for _, Child in pairs(ItemTpScroll:GetChildren()) do
			if Child:IsA("TextButton") then
				Child:Destroy()
			end
		end

		for _, Tool in pairs(workspace.Tools:GetChildren()) do
			local NewButton = Sample:Clone()
			NewButton.Parent = ItemTpScroll
			NewButton.Name = Tool.Name
			NewButton.Text = string.split(Tool.Name, "#")[1]
			NewButton.PositionTool.Value = Tool.Handle.Position
			if table.find(Utilities, string.split(Tool.Name, "#")[1]) then
				NewButton.BackgroundColor3 = Color3.fromRGB(149,149,149)
			elseif table.find(Rare, string.split(Tool.Name, "#")[1]) then
				NewButton.BackgroundColor3 = Color3.fromRGB(149, 0, 0)
			elseif table.find(Military, string.split(Tool.Name, "#")[1]) then
				NewButton.BackgroundColor3 = Color3.fromRGB(0, 149, 0)
			elseif table.find(Advanced, string.split(Tool.Name, "#")[1]) then
				NewButton.BackgroundColor3 = Color3.fromRGB(149, 0, 149)
			elseif table.find(Energy, string.split(Tool.Name, "#")[1]) then
				NewButton.BackgroundColor3 = Color3.fromRGB(0, 149, 149)
			elseif table.find(Artifacts, string.split(Tool.Name, "#")[1]) then
				NewButton.BackgroundColor3 = Color3.fromRGB(149, 149, 0)
			elseif table.find(Goggles, string.split(Tool.Name, "#")[1]) then
				NewButton.BackgroundColor3 = Color3.fromRGB(149, 66, 24)
			elseif table.find(MiscTools, string.split(Tool.Name, "#")[1]) then
				NewButton.BackgroundColor3 = Color3.fromRGB(149, 0, 149)
			end
			NewButton.Visible = true
			NewButton.MouseButton1Click:Connect(function()
				game.Players.LocalPlayer.Character.HumanoidRootPart.CFrame = CFrame.new(NewButton.PositionTool.Value, Vector3.new(0,0,0))
			end)
		end
	end
